CREATE TABLE products (
    product_no integer,
    name text,
    price numeric CHECK (price > 0),
    discounted_price numeric CHECK (discounted_price > 0),
    CHECK (price > discounted_price)
);

    population      float,
    altitude        int     -- in feet
);

    state           char(2)
) INHERITS (cities);
